public static void SelectionSort(List<int> values)
    {
        for (int i = 0; i < values.Count; i++)
        {
            int minIndex = i;
            for (int j = i + 1; j < values.Count; j++)
            {
                if (values[j] < values[minIndex])
                    minIndex = j;
            }
            (values[i], values[minIndex]) = (values[minIndex], values[i]);
        }
    }
