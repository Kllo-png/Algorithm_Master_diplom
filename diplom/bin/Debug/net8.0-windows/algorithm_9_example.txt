int jumpSearch(int integers[], int size, int elementToSearch) {
    int arrayLength = size;
    int jumpStep = (int) sqrt(arrayLength);
    int previousStep = 0;

    while (integers[std::min(jumpStep, arrayLength) - 1] < elementToSearch) {
        previousStep = jumpStep;
        jumpStep += (int)(sqrt(arrayLength));
        if (previousStep >= arrayLength)
            return -1;
    }
    
    while (integers[previousStep] < elementToSearch) {
        previousStep++;
        if (previousStep == std::min(jumpStep, arrayLength))
            return -1;
    }

    if (integers[previousStep] == elementToSearch)
        return previousStep;
    return -1;
}
